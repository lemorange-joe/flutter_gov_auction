FetchHistory
 - fetch_id, int PK, inc
 - url_en, varchar(500)	(e.g. https://www.gld.gov.hk/datagovhk/supplies-mgmt/auctionResult_2-2022_EN.csv)
 - url_tc, varchar(500)
 - url_sc, varchar(500)
 - file_path_en, varchar(300) (e.g. /var/data_files/csv/20220714-143000-auctionResult_2-2022_EN.csv)
 - file_path_tc, varchar(300)
 - file_path_sc, varchar(300)
 - status, char(1) (S: success, F: failed)
 - fetch_date, datetime

-----------

Location
 - location_id
 - address_en, varchar(500)
 - address_tc, varchar(300)
 - address_sc, varchar(300)

ItemType
	- type_id, int
	- code, varchar(2)
	- description_en, varchar(100)
	- description_tc, varchar(50)
	- description_sc, varchar(50)
	
Auction
	- auction_id, int PK, inc
	- auction_date, datetime
	- auction_num, varchar(20)
	- location_id, int
	- result_pdf_en, varchar(200)
	- result_pdf_tc, varchar(200)
	- result_pdf_sc, varchar(200)
	- auction_status, char(1) (P: pending, C: confirmed, X: cancelled, F: finished)
	- status, char(1) (A: active, I: inactive)
	- last_update, datetime

AuctionListPdf
	- pdf_id, int PK, inc 
	- auction_id, int
	- type_id, int
	- seq, int
	- url_en, varchar(200)
	- url_tc, varchar(200)
	- url_sc, varchar(200)
	
AuctionLot
	- lot_id, int PK, inc
	- auction_id, int
	- type_id, int
	- icon, varchar(100) (e.g. "fontawesome.car")
	- photo_url, varchar(200)
	- photo_real, bool
	- transaction_currency, char(3)
	- transaction_price, double
	- transaction_status, char(1) (S: sold, N: not-sold)
	- status, char(1) (A: active, I: inactive)
	- last_update, datetime
	
AuctionItem
	- item_id, int PK, inc
	- lot_id, int
	- seq, int
	- icon, varchar(100) (e.g. "fontawesome.car")
	- description_en, text
	- description_tc, text
	- description_sc, text
	- quantity, varchar(10)
	- unit_en, varchar(50)
	- unit_tc, varchar(5)
	- unit_sc, varchar(5)
